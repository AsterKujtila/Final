"use client";
import { useState, useRef, useEffect } from "react";
import { SellrThinking } from "@/components/sellr-thinking";
import { ArrowUp } from "lucide-react";
import { motion } from "framer-motion";

type Message = { role: "user" | "assistant"; content: string };

export default function ChatPage() {
  const [messages, setMessages] = useState<Message[]>([
    { role: "assistant", content: "Hey Seller 👋 What part of your Whop store is bleeding revenue today?" }
  ]);
  const[input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    scrollRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, isLoading]);

  const sendMessage = async () => {
    if (!input.trim() || isLoading) return;
    const userMsg = input;
    setInput("");
    setMessages((p) => [...p, { role: "user", content: userMsg }]);
    setIsLoading(true);

    try {
      // API call drejt Anthropic Claude
      const res = await fetch("/api/chat", {
        method: "POST",
        body: JSON.stringify({ messages:[...messages, { role: "user", content: userMsg }] }),
      });
      const data = await res.json();
      setMessages((p) => [...p, { role: "assistant", content: data.text }]);
    } catch (e) {
      console.error(e);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="flex flex-col h-screen max-w-4xl mx-auto pt-8 pb-6 px-4 md:px-8">
      
      <div className="flex-1 overflow-y-auto no-scrollbar space-y-6 pb-20">
        {messages.map((msg, idx) => (
          <motion.div
            key={idx}
            initial={{ opacity: 0, y: 10, scale: 0.98 }}
            animate={{ opacity: 1, y: 0, scale: 1 }}
            className={`flex ${msg.role === "user" ? "justify-end" : "justify-start"}`}
          >
            <div className={`max-w-[85%] rounded-[24px] px-6 py-4 font-sans text-[15px] leading-relaxed shadow-sm-custom ${
              msg.role === "user" 
                ? "bg-brown text-white rounded-br-sm" 
                : "bg-white border border-border text-brown rounded-tl-sm"
            }`}>
              {msg.content}
            </div>
          </motion.div>
        ))}
        {isLoading && <SellrThinking />}
        <div ref={scrollRef} />
      </div>

      <div className="relative mt-auto">
        <div className="absolute inset-x-0 bottom-0 bg-gradient-to-t from-cream to-transparent h-12 pointer-events-none -translate-y-full" />
        <div className="bg-white rounded-[24px] border border-border p-2 shadow-lg-custom flex items-end gap-2 transition-all focus-within:border-terracotta/50 focus-within:shadow-glow-terra">
          <textarea
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMessage(); }
            }}
            placeholder="Ask me anything..."
            className="flex-1 bg-transparent resize-none outline-none px-4 py-3 font-sans text-brown placeholder:text-brown-light max-h-32 min-h-[50px] no-scrollbar"
          />
          <button
            onClick={sendMessage}
            disabled={!input.trim() || isLoading}
            className="bg-terracotta hover:bg-terracotta-light disabled:opacity-50 text-white w-12 h-12 rounded-full flex items-center justify-center shrink-0 transition-colors"
          >
            <ArrowUp size={22} strokeWidth={2.5} />
          </button>
        </div>
        <p className="text-center font-mono text-[10px] text-brown-light mt-3 uppercase tracking-widest">
          SELLR uses 1 credit per message
        </p>
      </div>

    </div>
  );
}
